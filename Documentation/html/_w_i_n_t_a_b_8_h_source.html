00083 #ifndef NOFIX32
00084         #define INT(x)  HIWORD(x)
00085         #define FRAC(x) LOWORD(x)
00086 
00087         #define CASTFIX32(x)    ((FIX32)((x)*65536L))
00088 
00089         #define ROUND(x)                (INT(x) + (FRAC(x) > (WORD)0x8000))
00090 
00091         #define FIX_MUL(c, a, b)                                                \
00092                 (c = (((DWORD)FRAC(a) * FRAC(b)) >> 16) +       \
00093                         (DWORD)INT(a) * FRAC(b) +                               \
00094                         (DWORD)INT(b) * FRAC(a) +                               \
00095                         ((DWORD)INT(a) * INT(b) << 16))
00096 
00097         #ifdef _WINDLL
00098                 #define FIX_DIV_SC static
00099         #else
00100                 #define FIX_DIV_SC
00101         #endif
00102 
00103         #define FIX_DIV(c, a, b)                                        \
00104                 {                                                                               \
00105                         FIX_DIV_SC DWORD temp, rem, btemp;      \
00106                                                                                                 \
00107                         /* fraction done bytewise */            \
00108                         temp = ((a / b) << 16);                         \
00109                         rem = a % b;                                            \
00110                         btemp = b;                                                      \
00111                         if (INT(btemp) < 256) {                         \
00112                                 rem <<= 8;                                              \
00113                         }                                                                       \
00114                         else {                                                          \
00115                                 btemp >>= 8;                                    \
00116                         }                                                                       \
00117                         temp += ((rem / btemp) << 8);           \
00118                         rem %= btemp;                                           \
00119                         rem <<= 8;                                                      \
00120                         temp += rem / btemp;                            \
00121                         c = temp;                                                       \
00122                 }
