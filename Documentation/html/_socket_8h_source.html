00041 /*
00042  * Commands for ioctlsocket(),  taken from the BSD file fcntl.h.
00043  *
00044  *
00045  * Ioctl's have the command encoded in the lower word,
00046  * and the size of any in or out parameters in the upper
00047  * word.  The high 2 bits of the upper word are used
00048  * to encode the in/out status of the parameter; for now
00049  * we restrict parameters to at most 128 bytes.
00050  */
00051 #define IOCPARM_MASK    0x7f            /* parameters must be < 128 bytes */
00052 #define IOC_VOID        0x20000000      /* no parameters */
00053 #define IOC_OUT         0x40000000      /* copy out parameters */
00054 #define IOC_IN          0x80000000      /* copy in parameters */
00055 #define IOC_INOUT       (IOC_IN|IOC_OUT)
00056                                         /* 0x20000000 distinguishes new &
00057                                            old ioctl's */
00058 #define _IO(x,y)        (IOC_VOID|((x)<<8)|(y))
00059 
00060 #define _IOR(x,y,t)     (IOC_OUT|(((long)sizeof(t)&IOCPARM_MASK)<<16)|((x)<<8)|(y))
00061 
00062 #define _IOW(x,y,t)     (IOC_IN|(((long)sizeof(t)&IOCPARM_MASK)<<16)|((x)<<8)|(y))
00063 
00064 #define FIONREAD    _IOR('f', 127, u_long) /* get # bytes to read */
00065 #define FIONBIO     _IOW('f', 126, u_long) /* set/clear non-blocking i/o */
00066 #define FIOASYNC    _IOW('f', 125, u_long) /* set/clear async i/o */
