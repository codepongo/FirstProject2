00332 template< typename TYPE >
00333 HRESULT CGrowableArray<TYPE>::SetSizeInternal( int nNewMaxSize )
00334 {
00335     if( nNewMaxSize < 0 || ( nNewMaxSize > INT_MAX / sizeof( TYPE ) ) )
00336     {
00337         assert( false );
00338         return E_INVALIDARG;
00339     }
00340 
00341     if( nNewMaxSize == 0 )
00342     {
00343         // Shrink to 0 size & cleanup
00344         if( m_pData )
00345         {
00346             free( m_pData );
00347             m_pData = NULL;
00348         }
00349 
00350         m_nMaxSize = 0;
00351         m_nSize = 0;
00352     }
00353     else if( m_pData == NULL || nNewMaxSize > m_nMaxSize )
00354     {
00355         // Grow array
00356         int nGrowBy = ( m_nMaxSize == 0 ) ? 16 : m_nMaxSize;
00357 
00358         // Limit nGrowBy to keep m_nMaxSize less than INT_MAX
00359         if( (UINT)m_nMaxSize + (UINT)nGrowBy > (UINT)INT_MAX )
00360             nGrowBy = INT_MAX - m_nMaxSize;
00361 
00362         nNewMaxSize = __max( nNewMaxSize, m_nMaxSize + nGrowBy );
00363 
00364         // Verify that (nNewMaxSize * sizeof(TYPE)) is not greater than UINT_MAX or the realloc will overrun
00365         if( sizeof(TYPE) > UINT_MAX / (UINT)nNewMaxSize )
00366             return E_INVALIDARG;
00367 
00368         TYPE* pDataNew = (TYPE*) realloc( m_pData, nNewMaxSize * sizeof(TYPE) );
00369         if( pDataNew == NULL )
00370             return E_OUTOFMEMORY;
00371 
00372         m_pData = pDataNew;
00373         m_nMaxSize = nNewMaxSize;
00374     }
00375 
00376     return S_OK;
00377 }
